struct Student = {
  age: Int,
  part_time: Bool,
  avg_grade: Float
};

def something(x: Int): Student = {
  Student(x, true, 75.0)
};

def main() = {
  val st: Student = Student(20, false, 89.0);
  val ret: Student = something(20);
  print(st.age);
  print(st.part_time);
  print(st.avg_grade);
  print(ret.age);
  print(ret.part_time);
  print(ret.avg_grade);
  0
}
